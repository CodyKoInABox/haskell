isLeapYear :: Integer -> Bool
isLeapYear year =
  year `rem` 4 == 0 && year `rem` 100 /= 0 ||
  year `rem` 4 == 0 && year `rem` 400 == 0

main :: IO()
main = do
  print(isLeapYear 1996)

factorial.hs

Run

factorial :: Int -> Int
factorial x =
  if x <= 1 then
   1
  else
   x * factorial (x - 1)

main :: IO()
main = do
  print(factorial 5)

inRange :: Int -> Int -> Int -> Bool
inRange min max x =
  x >= min && x <= max

main :: IO()
main = do
  print(inRange 5 15 10)
